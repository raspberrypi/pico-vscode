72static inline uint32_t us_to_ms(uint64_t us) {
73 if (us >> 32u) {
74 return (uint32_t)(us / 1000u);
75 } else {
76 return ((uint32_t)us) / 1000u;
77 }
78}

111static inline absolute_time_t delayed_by_us(const absolute_time_t t, uint64_t us) {
112 absolute_time_t t2;
113 uint64_t base = to_us_since_boot(t);
114 uint64_t delayed = base + us;
115 if ((int64_t)delayed < 0) {
116 // absolute_time_t (to allow for signed time deltas) is never greater than INT64_MAX which == at_the_end_of_time
117 delayed = INT64_MAX;
118 }
119 update_us_since_boot(&t2, delayed);
120 return t2;
121}
122
130static inline absolute_time_t delayed_by_ms(const absolute_time_t t, uint32_t ms) {
131 absolute_time_t t2;
132 uint64_t base = to_us_since_boot(t);
133 uint64_t delayed = base + ms * 1000ull;
134 if ((int64_t)delayed < 0) {
135 // absolute_time_t (to allow for signed time deltas) is never greater than INT64_MAX which == at_the_end_of_time
136 delayed = INT64_MAX;
137 }
138 update_us_since_boot(&t2, delayed);
139 return t2;
140}
