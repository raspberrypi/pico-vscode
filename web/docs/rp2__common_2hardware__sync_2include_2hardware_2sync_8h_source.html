246__force_inline static uint32_t save_and_disable_interrupts(void) {
247 uint32_t status;
248#ifdef __riscv
249 pico_default_asm_volatile (
250 "csrrci %0, mstatus, 0x8\n"
251 : "=r" (status) :: "memory"
252 );
253#else
254 pico_default_asm_volatile (
255 "mrs %0, PRIMASK\n"
256 "cpsid i"
257 : "=r" (status) :: "memory");
258#endif
259 return status;
260}

267__force_inline static void restore_interrupts(uint32_t status) {
268#ifdef __riscv
269 __compiler_memory_barrier();
270 if (status & 0x8) {
271 riscv_set_csr(mstatus, 8);
272 } else {
273 riscv_clear_csr(mstatus, 8);
274 }
275 __compiler_memory_barrier();
276#else
277 pico_default_asm_volatile ("msr PRIMASK,%0"::"r" (status) : "memory" );
278#endif
279}

289__force_inline static void restore_interrupts_from_disabled(uint32_t status) {
290#ifdef __riscv
291 // on RISC-V this can enable interrupts, but not disable interrupts... which
292 // is the common case and doesn't require a branch
293 __compiler_memory_barrier();
294 riscv_set_csr(mstatus, status & 8);
295 __compiler_memory_barrier();
296#else
297 // on ARM, this behaves the same as restore_interrupts()
298 pico_default_asm_volatile ("msr PRIMASK,%0"::"r" (status) : "memory" );
299#endif
300}
