133static inline void interp_config_set_shift(interp_config *c, uint shift) {
134 valid_params_if(HARDWARE_INTERP, shift < 32);
135 c->ctrl = (c->ctrl & ~SIO_INTERP0_CTRL_LANE0_SHIFT_BITS) |
136 ((shift << SIO_INTERP0_CTRL_LANE0_SHIFT_LSB) & SIO_INTERP0_CTRL_LANE0_SHIFT_BITS);
137}

148static inline void interp_config_set_mask(interp_config *c, uint mask_lsb, uint mask_msb) {
149 valid_params_if(HARDWARE_INTERP, mask_msb < 32);
150 valid_params_if(HARDWARE_INTERP, mask_lsb <= mask_msb);
151 c->ctrl = (c->ctrl & ~(SIO_INTERP0_CTRL_LANE0_MASK_LSB_BITS | SIO_INTERP0_CTRL_LANE0_MASK_MSB_BITS)) |
152 ((mask_lsb << SIO_INTERP0_CTRL_LANE0_MASK_LSB_LSB) & SIO_INTERP0_CTRL_LANE0_MASK_LSB_BITS) |
153 ((mask_msb << SIO_INTERP0_CTRL_LANE0_MASK_MSB_LSB) & SIO_INTERP0_CTRL_LANE0_MASK_MSB_BITS);
154}

257static inline void interp_config_set_force_bits(interp_config *c, uint bits) {
258 invalid_params_if(HARDWARE_INTERP, bits > 3);
259 // note cannot use hw_set_bits on SIO
260 c->ctrl = (c->ctrl & ~SIO_INTERP0_CTRL_LANE0_FORCE_MSB_BITS) |
261 (bits << SIO_INTERP0_CTRL_LANE0_FORCE_MSB_LSB);
262}

287static inline void interp_set_config(interp_hw_t *interp, uint lane, interp_config *config) {
288 invalid_params_if(HARDWARE_INTERP, lane > 1);
289 invalid_params_if(HARDWARE_INTERP, config->ctrl & SIO_INTERP1_CTRL_LANE0_CLAMP_BITS &&
290 (!interp_index(interp) || lane)); // only interp1 lane 0 has clamp bit
291 invalid_params_if(HARDWARE_INTERP, config->ctrl & SIO_INTERP0_CTRL_LANE0_BLEND_BITS &&
292 (interp_index(interp) || lane)); // only interp0 lane 0 has blend bit
293 interp->ctrl[lane] = config->ctrl;
294}
