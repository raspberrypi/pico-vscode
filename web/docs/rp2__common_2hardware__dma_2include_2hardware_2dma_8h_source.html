185#define DMA_ADDRESS_UPDATE_TYPE_TO_DMA_CH_CTRL_WRITE_BITS(u) \
186((((u)&1) << DMA_CH0_CTRL_TRIG_INCR_WRITE_LSB) | \

310static inline void channel_config_set_transfer_data_size(dma_channel_config_t *c, dma_channel_transfer_size_t size) {

330static inline void channel_config_set_ring(dma_channel_config_t *c, bool write, uint size_bits) {
331 assert(size_bits < 32);
332 c->ctrl = (c->ctrl & ~(DMA_CH0_CTRL_TRIG_RING_SIZE_BITS | DMA_CH0_CTRL_TRIG_RING_SEL_BITS)) |
333 (size_bits << DMA_CH0_CTRL_TRIG_RING_SIZE_LSB) |
334 (write ? DMA_CH0_CTRL_TRIG_RING_SEL_BITS : 0);
335}

430static inline dma_channel_config_t dma_channel_get_default_config(uint channel) {
431 dma_channel_config_t c = {0};
432 channel_config_set_read_increment(&c, true);
433 channel_config_set_write_increment(&c, false);
434 channel_config_set_dreq(&c, DREQ_FORCE);
435 channel_config_set_chain_to(&c, channel);
436 channel_config_set_transfer_data_size(&c, DMA_SIZE_32);

759static inline void dma_channel_abort(uint channel) {
760 check_dma_channel_param(channel);
761 dma_hw->abort = 1u << channel;
762 // Bit will go 0 once channel has reached safe state
763 // (i.e. any in-flight transfers have retired)
764 while (dma_hw->ch[channel].ctrl_trig & DMA_CH0_CTRL_TRIG_BUSY_BITS) tight_loop_contents();

966inline static void dma_sniffer_enable(uint channel, uint mode, bool force_channel_enable) {
967 check_dma_channel_param(channel);
968 check_hw_layout(dma_hw_t, sniff_ctrl, DMA_SNIFF_CTRL_OFFSET);
969 if (force_channel_enable) {
970 hw_set_bits(&dma_hw->ch[channel].al1_ctrl, DMA_CH0_CTRL_TRIG_SNIFF_EN_BITS);
971 }
972 hw_write_masked(&dma_hw->sniff_ctrl,
973 (((channel << DMA_SNIFF_CTRL_DMACH_LSB) & DMA_SNIFF_CTRL_DMACH_BITS) |
974 ((mode << DMA_SNIFF_CTRL_CALC_LSB) & DMA_SNIFF_CTRL_CALC_BITS) |
975 DMA_SNIFF_CTRL_EN_BITS),
976 (DMA_SNIFF_CTRL_DMACH_BITS |
977 DMA_SNIFF_CTRL_CALC_BITS |
978 DMA_SNIFF_CTRL_EN_BITS));
979}
