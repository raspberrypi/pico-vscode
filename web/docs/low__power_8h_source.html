364static inline bool low_power_start_aon_timer_at_time_ms(uint64_t ms) {
365 struct timespec ts;
366 ms_to_timespec(ms, &ts);
367 return aon_timer_start(&ts);
368}

379static inline bool low_power_start_aon_timer(void) {
380 if (aon_timer_is_running()) return true;
381 return low_power_start_aon_timer_at_time_ms(0);
382}
