81#define fixed_bitset_init(ptr, type, N, fill) ({ \
82 assert(sizeof(type) == fixed_bitset_sizeof_for(N)); \
83 __unused type *type_check = ptr; \
84 (ptr)->bitset.size = N; \
85 (ptr)->bitset.word_size = ((N) + 31u) / 32u; \
86 __builtin_memset(&(ptr)->bitset.words, (fill) ? 0xff : 0, (ptr)->bitset.word_size * sizeof(uint32_t)); \
