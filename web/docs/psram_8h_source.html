248#define psram_or_malloc(group, type, var, size) \
249 static type __uninitialized_psram(group) var##_psram[size]; static type* var; \
250 if (!var) { \
251 if (psram_check_address(var##_psram + (size))) { \
252 var = (type*)var##_psram; \
253 } else { \
254 var = (type*)malloc((size) * sizeof(type)); \
255 } \
256 }
