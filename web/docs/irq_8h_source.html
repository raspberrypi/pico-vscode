472static inline void irq_clear(uint int_num) {
473#if PICO_RP2040
474 *((volatile uint32_t *) (PPB_BASE + M0PLUS_NVIC_ICPR_OFFSET)) = (1u << ((uint32_t) (int_num & 0x1F)));
475#elif defined(__riscv)
476 // External IRQs are not latched, but we should clear the IRQ force bit here
477 hazard3_irqarray_clear(RVCSR_MEIFA_OFFSET, int_num / 16, 1u << (int_num % 16));
478#else
479 nvic_hw->icpr[int_num/32] = 1 << (int_num % 32);
480#endif
481}
