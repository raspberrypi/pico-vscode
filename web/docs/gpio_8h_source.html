561static inline uint32_t gpio_get_irq_event_mask(uint gpio) {
562 check_gpio_param(gpio);
563 io_bank0_irq_ctrl_hw_t *irq_ctrl_base = get_core_irq_ctrl(get_core_num());
564 io_ro_32 *status_reg = &irq_ctrl_base->ints[gpio >> 3u];
565 return (*status_reg >> (4 * (gpio & 7u))) & 0xfu;
566}

1180static inline void gpio_put(uint gpio, bool value) {
1181#if PICO_USE_GPIO_COPROCESSOR
1182 gpioc_bit_out_put(gpio, value);
1183#elif NUM_BANK0_GPIOS <= 32
1184 uint32_t mask = 1ul << gpio;
1185 if (value)
1186 gpio_set_mask(mask);
1187 else
1188 gpio_clr_mask(mask);
1189#else
1190 uint32_t mask = 1ul << (gpio & 0x1fu);
1191 if (gpio < 32) {
1192 if (value) {
1193 sio_hw->gpio_set = mask;
1194 } else {
1195 sio_hw->gpio_clr = mask;
1196 }
1197 } else {
1198 if (value) {
1199 sio_hw->gpio_hi_set = mask;
1200 } else {
1201 sio_hw->gpio_hi_clr = mask;
1202 }
1203 }
1204#endif
1205}
