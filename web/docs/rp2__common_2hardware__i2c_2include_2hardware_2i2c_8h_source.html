427static inline void i2c_write_raw_blocking(i2c_inst_t *i2c, const uint8_t *src, size_t len) {
428 for (size_t i = 0; i < len; ++i) {
429 // TODO NACK or STOP on end?
430 while (!i2c_get_write_available(i2c))
431 tight_loop_contents();
432 i2c_get_hw(i2c)->data_cmd = *src++;
433 }
434}
435
446static inline void i2c_read_raw_blocking(i2c_inst_t *i2c, uint8_t *dst, size_t len) {
447 for (size_t i = 0; i < len; ++i) {
448 while (!i2c_get_read_available(i2c))
449 tight_loop_contents();
450 *dst++ = (uint8_t)i2c_get_hw(i2c)->data_cmd;
451 }
452}

463static inline uint8_t i2c_read_byte_raw(i2c_inst_t *i2c) {
464 i2c_hw_t *hw = i2c_get_hw(i2c);
465 assert(hw->status & I2C_IC_STATUS_RFNE_BITS); // Rx FIFO must not be empty
466 return (uint8_t)hw->data_cmd;
467}

478static inline void i2c_write_byte_raw(i2c_inst_t *i2c, uint8_t value) {
479 i2c_hw_t *hw = i2c_get_hw(i2c);
480 assert(hw->status & I2C_IC_STATUS_TFNF_BITS); // Tx FIFO must not be full
481 hw->data_cmd = value;
482}
