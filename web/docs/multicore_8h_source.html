189static inline void multicore_fifo_push_blocking_inline(uint32_t data) {
190 // We wait for the fifo to have some space
191 while (!multicore_fifo_wready())
192 tight_loop_contents();
193
194 sio_hw->fifo_wr = data;
195
196 // Fire off an event to the other core
197 __sev();
198}
