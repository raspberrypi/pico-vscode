197static inline bool status_led_set_state(bool led_on) {
198 if (status_led_via_colored_status_led()) {
199 return colored_status_led_set_state(led_on);
200 } else if (status_led_supported()) {
201#if defined(PICO_DEFAULT_LED_PIN)
202 #if PICO_DEFAULT_LED_PIN_INVERTED
203 gpio_put(PICO_DEFAULT_LED_PIN, !led_on);
204 #else
205 gpio_put(PICO_DEFAULT_LED_PIN, led_on);
206 #endif
207 return true;
208#elif defined(CYW43_WL_GPIO_LED_PIN)
209 cyw43_gpio_set(&cyw43_state, CYW43_WL_GPIO_LED_PIN, led_on);
210 return true;
211#endif
212 }
213 return false;
214}
215
223static inline bool status_led_get_state() {
224 if (status_led_via_colored_status_led()) {
225 return colored_status_led_get_state();
226 } else if (status_led_supported()) {
227#if defined(PICO_DEFAULT_LED_PIN)
228 #if PICO_DEFAULT_LED_PIN_INVERTED
229 return !gpio_get(PICO_DEFAULT_LED_PIN);
230 #else
231 return gpio_get(PICO_DEFAULT_LED_PIN);
232 #endif
233#elif defined CYW43_WL_GPIO_LED_PIN
234 bool value = false;
235 cyw43_gpio_get(&cyw43_state, CYW43_WL_GPIO_LED_PIN, &value);
236 return value;
237#endif
238 }
239 return false;
240}
