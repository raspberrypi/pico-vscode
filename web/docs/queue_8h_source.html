92static inline uint queue_get_level_unsafe(queue_t *q) {
93 int32_t rc = (int32_t)q->wptr - (int32_t)q->rptr;
94 if (rc < 0) {
95 rc += q->element_count + 1;
96 }
97 return (uint)rc;
98}

106static inline uint queue_get_level(queue_t *q) {
107 uint32_t save = spin_lock_blocking(q->core.spin_lock);
108 uint level = queue_get_level_unsafe(q);
109 spin_unlock(q->core.spin_lock, save);
110 return level;
111}
