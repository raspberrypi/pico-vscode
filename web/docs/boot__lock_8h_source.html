53__force_inline static uint boot_lock_get_num(boot_lock_t *lock) {
54 invalid_params_if(HARDWARE_BOOT_LOCK, (uint) lock < BOOTRAM_BASE + BOOTRAM_BOOTLOCK0_OFFSET ||
55 (uint) lock >= NUM_BOOT_LOCKS * sizeof(boot_lock_t) + BOOTRAM_BASE + BOOTRAM_BOOTLOCK0_OFFSET ||
56 ((uint) lock - BOOTRAM_BASE + BOOTRAM_BOOTLOCK0_OFFSET) % sizeof(boot_lock_t) != 0);
57 return (uint) (lock - (boot_lock_t *) (BOOTRAM_BASE + BOOTRAM_BOOTLOCK0_OFFSET));
58}

65__force_inline static void boot_lock_unsafe_blocking(boot_lock_t *lock) {
66 // Note we don't do a wfe or anything, because by convention these boot_locks are VERY SHORT LIVED and NEVER BLOCK and run
67 // with INTERRUPTS disabled (to ensure that)... therefore nothing on our core could be blocking us, so we just need to wait on another core
68 // anyway which should be finished soon
69 while (__builtin_expect(!*lock, 0)) { // read from bootlock register (tries to acquire the lock)
70 tight_loop_contents();
71 }
72 __mem_fence_acquire();
73}
74
80__force_inline static bool boot_try_lock_unsafe(boot_lock_t *lock) {
81 if (*lock) {
82 __mem_fence_acquire();
83 return true;
84 }
85 return false;
86}

93__force_inline static void boot_unlock_unsafe(boot_lock_t *lock) {
94 __mem_fence_release();
95 *lock = 0; // write to bootlock register (release lock)
96}
97
106__force_inline static uint32_t boot_lock_blocking(boot_lock_t *lock) {
107 uint32_t save = save_and_disable_interrupts();
108 boot_lock_unsafe_blocking(lock);
109 return save;
110}

117inline static bool is_boot_locked(boot_lock_t *lock) {
118 check_hw_size(boot_lock_t, 4);
119 uint lock_num = boot_lock_get_num(lock);
120 return 0 != (*(io_ro_32 *) (BOOTRAM_BASE + BOOTRAM_BOOTLOCK_STAT_OFFSET) & (1u << lock_num));
121}

133__force_inline static void boot_unlock(boot_lock_t *lock, uint32_t saved_irq) {
