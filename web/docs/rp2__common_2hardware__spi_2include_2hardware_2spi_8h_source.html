282static inline void spi_set_format(spi_inst_t *spi, uint data_bits, spi_cpol_t cpol, spi_cpha_t cpha, __unused spi_order_t order) {
283 valid_params_if(HARDWARE_SPI, SPI_IS_INSTANCE(spi));
284 invalid_params_if(HARDWARE_SPI, data_bits < 4 || data_bits > 16);
285 // LSB-first not supported on PL022:
286 invalid_params_if(HARDWARE_SPI, order != SPI_MSB_FIRST);
287 invalid_params_if(HARDWARE_SPI, cpol != SPI_CPOL_0 && cpol != SPI_CPOL_1);
288 invalid_params_if(HARDWARE_SPI, cpha != SPI_CPHA_0 && cpha != SPI_CPHA_1);
289
290 // Disable the SPI
291 uint32_t enable_mask = spi_get_hw(spi)->cr1 & SPI_SSPCR1_SSE_BITS;
292 hw_clear_bits(&spi_get_hw(spi)->cr1, SPI_SSPCR1_SSE_BITS);
293
294 hw_write_masked(&spi_get_hw(spi)->cr0,
295 ((uint)(data_bits - 1)) << SPI_SSPCR0_DSS_LSB |
296 ((uint)cpol) << SPI_SSPCR0_SPO_LSB |
297 ((uint)cpha) << SPI_SSPCR0_SPH_LSB,
298 SPI_SSPCR0_DSS_BITS |
299 SPI_SSPCR0_SPO_BITS |
300 SPI_SSPCR0_SPH_BITS);
301
302 // Re-enable the SPI
303 hw_set_bits(&spi_get_hw(spi)->cr1, enable_mask);
304}

315static inline void spi_set_slave(spi_inst_t *spi, bool slave) {
316 valid_params_if(HARDWARE_SPI, SPI_IS_INSTANCE(spi));
317 // Disable the SPI
318 uint32_t enable_mask = spi_get_hw(spi)->cr1 & SPI_SSPCR1_SSE_BITS;
319 hw_clear_bits(&spi_get_hw(spi)->cr1, SPI_SSPCR1_SSE_BITS);
320
321 if (slave)
322 hw_set_bits(&spi_get_hw(spi)->cr1, SPI_SSPCR1_MS_BITS);
323 else
324 hw_clear_bits(&spi_get_hw(spi)->cr1, SPI_SSPCR1_MS_BITS);
325
326 // Re-enable the SPI
327 hw_set_bits(&spi_get_hw(spi)->cr1, enable_mask);
328}
