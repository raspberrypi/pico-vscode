93static inline void busy_wait_at_least_cycles(uint32_t minimum_cycles) {
94 pico_default_asm_volatile (
95#ifdef __riscv
96 // Note the range is halved on RISC-V due to signed comparison (no carry flag)
97 ".option push\n"
98 ".option norvc\n" // force 32 bit addi, so branch prediction guaranteed
99 ".p2align 2\n"
100 "1: \n"
101 "addi %0, %0, -2 \n"
102 "bgez %0, 1b\n"
103 ".option pop"
104#else
105 "1: subs %0, #3\n"
106 "bcs 1b\n"
107#endif
108 : "+r" (minimum_cycles) : : "cc", "memory"
109 );
110}

145static __force_inline uint __get_current_exception(void) {
146#ifdef __riscv
147 uint32_t meicontext;
148 pico_default_asm_volatile (
149 "csrr %0, %1\n"
150 : "=r" (meicontext) : "i" (RVCSR_MEICONTEXT_OFFSET)
151 );
152 if (meicontext & RVCSR_MEICONTEXT_NOIRQ_BITS) {
153 return 0;
154 } else {
155 return VTABLE_FIRST_IRQ + (
156 (meicontext & RVCSR_MEICONTEXT_IRQ_BITS) >> RVCSR_MEICONTEXT_IRQ_LSB
157 );
158 }
159#else
160 uint exception;
161 pico_default_asm_volatile (
162 "mrs %0, ipsr\n"
163 "uxtb %0, %0\n"
164 : "=l" (exception)
165 );
166 return exception;
167#endif
168}

175__force_inline static bool pico_processor_state_is_nonsecure(void) {
176#ifndef __riscv
177 // todo add a define to disable NS checking at all?
178 // IDAU-Exempt addresses return S=1 when tested in the Secure state,
179 // whereas executing a tt in the NonSecure state will always return S=0.
180 uint32_t tt;
181 pico_default_asm_volatile (
182 "movs %0, #0\n"
183 "tt %0, %0\n"
184 : "=r" (tt) : : "cc"
185 );
186 return !(tt & (1u << 22));
187#else
188 // NonSecure is an Arm concept, there is nothing meaningful to return
189 // here. Note it's not possible in general to detect whether you are
190 // executing in U-mode as, for example, M-mode is classically
191 // virtualisable in U-mode.
192 return false;
193#endif
194}

232#ifdef __riscv
